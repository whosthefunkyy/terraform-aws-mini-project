data "aws_iam_policy_document" "assume_role" {
  statement {
    effect = "Allow"
    principals {
      type        = "Service"
      identifiers = ["ec2.amazonaws.com"]
    }
    actions = ["sts:AssumeRole"]
  }
}

# 2. Создание самой IAM-роли
resource "aws_iam_role" "ssm_ec2_role" {
  name               = "SSM_EC2_Access_Role"
  assume_role_policy = data.aws_iam_policy_document.assume_role.json
}

# 3. Присоединение стандартной политики AWS для работы SSM
resource "aws_iam_role_policy_attachment" "ssm_core_attach" {
  role       = aws_iam_role.ssm_ec2_role.name
  # Это стандартная политика, разрешающая инстансу работать с SSM
  policy_arn = "arn:aws:iam::aws:policy/AmazonSSMManagedInstanceCore" 
}

# 4. Создание профиля инстанса для привязки роли к EC2
resource "aws_iam_instance_profile" "ssm_ec2_profile" {
  name = "SSM_EC2_Profile"
  role = aws_iam_role.ssm_ec2_role.name
}
